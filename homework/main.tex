
% right area of the page text aligned to the right of the box
\newcommand{\rightpagerightalignment}[2]{%
    \normalmarginpar{
        \marginpar{%
            \begin{minipage}{3cm}%
                \raggedleft%
                \textcolor{#1}{#2}%
            \end{minipage}%
        }%
    }
}

\newcommand{\leftpage}[3]{%
    \hspace*{-\textwidth}\begin{minipage}{3cm}%
        \vspace*{#1}%
        \textcolor{#2}{#3}%
    \end{minipage}%
}
